# TSan suppression file for known false positives on gcc 10.3 / aarch64.
#
# WARNING: these suppressions are broad (match by function name pattern).
# They hide ALL races in the matched functions, not just ThreadPool taskQ_
# false positives. Before adding a new rule, verify the race is a true
# false positive (synchronized via mutex but tsan doesn't see it).
#
# If a real race is introduced in these functions in the future, it will
# be hidden. Periodically review by temporarily removing suppressions
# and checking if new races appear.

# 1. std::shared_ptr / packaged_task control block races
# False positive: make_shared writes control block, worker thread reads
# it via operator->(). Synchronized via ThreadPool mtx_ on taskQ_ push/pop,
# but gcc 10.3 tsan doesn't instrument std::queue's internal mutex.
race:std::__shared_ptr*
race:std::packaged_task*
race:std::__future_base*

# 2. TBB concurrent_hash_map internal bucket locking
# TBB uses custom spinlocks that tsan doesn't instrument. Races on
# search_bucket/add_to_bucket are internal to TBB's implementation.
race:tbb::interface5::concurrent_hash_map*
race:tbb::interface5::internal::hash_map_base*

# 3. std::deque / std::queue internal access (ThreadPool taskQ_)
# tsan doesn't see std::queue's internal std::deque mutex as sync point.
race:std::deque*
race:std::queue*
